# CosmiKAI — Python dependencies
# Install with: pip install -r requirements.txt

# ── Astronomy / data acquisition ───────────────────────────────────────────
lightkurve        # MAST light curve download (wraps astroquery)
astropy           # BoxLeastSquares BLS, time handling, units

# ── Machine learning ────────────────────────────────────────────────────────
torch             # TransitCNN 1D-CNN inference (CPU build sufficient for server)
numpy             # Array operations throughout the pipeline

# ── Signal processing ───────────────────────────────────────────────────────
scipy             # savgol_filter used in flatten_flux (preprocess.py)

# ── Backend API server ──────────────────────────────────────────────────────
fastapi           # REST API framework
uvicorn[standard] # ASGI server (includes websockets + fast C extensions)
aiosqlite         # Async SQLite driver used by the prediction cache

# ── Data validation ─────────────────────────────────────────────────────────
pydantic          # Request/response schema validation (bundled with FastAPI)
